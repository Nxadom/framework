5. Cara Penggunaan

<!-- Set tema di elemen root -->
<html data-color-mode="light" data-light-theme="light">
<html data-color-mode="dark" data-dark-theme="dark_dimmed">
<html data-color-mode="auto" data-light-theme="light" data-dark-theme="dark">

/* Gunakan CSS variables di stylesheet */
.my-element {
  background: var(--bgColor-accent-muted);
  color: var(--fgColor-accent);
  border: 1px solid var(--borderColor-accent-emphasis);
}
